理事長一次性設定：
  1. 開你的「工會名冊」Google 試算表
  2. 上方選單 → 擴充功能 → Apps Script
  3. 把下面整段程式碼加進 程式碼.gs（不要刪原有的）
  4. 右上「部署 → 新增部署」→ 類型選「網路應用程式」
  5. 執行身分選「我」、存取權選「任何人」→ 部署 → 複製網址
  6. 把網址貼到下方 → 儲存 → 傳給所有會員
複製貼進 Apps Script 的程式碼：
// ── 加班情境回報 doPost（加進 程式碼.gs）── function doPost(e) { try { var data = JSON.parse(e.postData.contents); var ss = SpreadsheetApp.getActiveSpreadsheet(); var sh = ss.getSheetByName('加班回報'); if (!sh) { sh = ss.insertSheet('加班回報'); sh.appendRow(['回報時間','加班日期','班別','姓名/匿名', '情境','緊急程度','加班分鐘','備註']); sh.setFrozenRows(1); } var ts = Utilities.formatDate(new Date(),'Asia/Taipei','yyyy-MM-dd HH:mm:ss'); var name = data.name || '（匿名）'; var note = data.note || ''; (data.items || []).forEach(function(item) { sh.appendRow([ts, data.date, data.shift, name, item.label, item.urgency, item.mins, note]); }); return ContentService .createTextOutput(JSON.stringify({ok:true})) .setMimeType(ContentService.MimeType.JSON); } catch(err) { return ContentService .createTextOutput(JSON.stringify({ok:false,error:err.toString()})) .setMimeType(ContentService.MimeType.JSON); } }
